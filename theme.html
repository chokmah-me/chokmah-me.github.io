Tailwind Config Snippet

module.exports = {
  theme: {
    extend: {
      colors: {
        chokmah: {
          base:      '#0f172a',
          surface:   '#1e293b',
          accent:    '#d97706',
          highlight: '#f59e0b',
          text:      '#f8fafc',
          muted:     '#cbd5e1',
        }
      }
    }
  }
}

Copy this into your tailwind.config.js to use bg-chokmah-base, text-chokmah-accent, etc.
